## Implementation
[EXTRACT_AUDIO_TRANSCRIPT](EXTRACT_AUDIO_TRANSCRIPT.orch.yaml):
This extracts data from the internally staged Audio `.mp3` Data: `XXXX.mp3`. Into the table `CALL_CENTER_ANALYTICS_DW.EXTRACT.AUDIO_FILES`

```sql
-- SQL Script Component
INSERT INTO "CALL_CENTER_ANALYTICS_DW"."EXTRACT"."AUDIO_FILES" (
    CALL_TRANSCRIPT_ID,
    FILE_PATH,
    AUDIO_FILE,
    AUDIO_FILE_URL,
    FILE_SIZE,
    LAST_MODIFIED,
    FILE_EXTENSION
)
SELECT 
    REPLACE(RELATIVE_PATH, '.mp3', '') AS CALL_TRANSCRIPT_ID,
    RELATIVE_PATH AS FILE_PATH,
    TO_FILE('@CALL_CENTER_ANALYTICS_DW.EXTRACT.INT_STAGE_DOC', RELATIVE_PATH) AS AUDIO_FILE,
    BUILD_STAGE_FILE_URL('@CALL_CENTER_ANALYTICS_DW.EXTRACT.INT_STAGE_DOC', RELATIVE_PATH) AS AUDIO_FILE_URL,
    SIZE AS FILE_SIZE,
    LAST_MODIFIED,
    SPLIT_PART(RELATIVE_PATH, '.', -1) AS FILE_EXTENSION
FROM 
    DIRECTORY('@CALL_CENTER_ANALYTICS_DW.EXTRACT.INT_STAGE_DOC')
WHERE 
    RELATIVE_PATH ILIKE '%.mp3'
    AND REPLACE(RELATIVE_PATH, '.mp3', '') NOT IN (
        SELECT CALL_TRANSCRIPT_ID 
        FROM "CALL_CENTER_ANALYTICS_DW"."EXTRACT"."AUDIO_FILES"
    );
```

[EXTRACT_LOOKUP](EXTRACT_LOOKUP.orch.yaml): 
This extracts data from the internally staged `json` Data: `LOOKUP.json`. Into the table `CALL_CENTER_ANALYTICS_DW.EXTRACT.LOOKUP`
```sql
INSERT INTO "CALL_CENTER_ANALYTICS_DW"."EXTRACT"."LOOKUP" (
    LOOKUP_FILTER,
    LOOKUP_ID,
    LOOKUP_CODE,
    LOOKUP_LOWER_BOUND,
    LOOKUP_UPPER_BOUND,
    LOOKUP_SHORT_NAME,
    LOOKUP_LONG_NAME,
    LOOKUP_DESC,
    LOOKUP_SORT_ORDER
)
WITH stage_data AS (
    SELECT 
        $1:LOOKUP_FILTER::VARCHAR AS LOOKUP_FILTER,
        $1:LOOKUP_ID::NUMBER AS LOOKUP_ID,
        $1:LOOKUP_CODE::VARCHAR AS LOOKUP_CODE,
        $1:LOOKUP_LOWER_BOUND::FLOAT AS LOOKUP_LOWER_BOUND,
        $1:LOOKUP_UPPER_BOUND::FLOAT AS LOOKUP_UPPER_BOUND,
        $1:LOOKUP_SHORT_NAME::VARCHAR AS LOOKUP_SHORT_NAME,
        $1:LOOKUP_LONG_NAME::VARCHAR AS LOOKUP_LONG_NAME,
        $1:LOOKUP_DESC::VARCHAR AS LOOKUP_DESC,
        $1:LOOKUP_SORT_ORDER::NUMBER AS LOOKUP_SORT_ORDER
    FROM 
        @"CALL_CENTER_ANALYTICS_DW"."EXTRACT"."INT_STAGE_DOC"/LOOKUP.json
        (FILE_FORMAT => '"CALL_CENTER_ANALYTICS_DW"."EXTRACT"."JSON_FORMAT"')
)
SELECT 
    s.LOOKUP_FILTER,
    s.LOOKUP_ID,
    s.LOOKUP_CODE,
    s.LOOKUP_LOWER_BOUND,
    s.LOOKUP_UPPER_BOUND,
    s.LOOKUP_SHORT_NAME,
    s.LOOKUP_LONG_NAME,
    s.LOOKUP_DESC,
    s.LOOKUP_SORT_ORDER
FROM stage_data s
WHERE NOT EXISTS (
    SELECT 1 
    FROM "CALL_CENTER_ANALYTICS_DW"."EXTRACT"."LOOKUP" t
    WHERE t.LOOKUP_FILTER = s.LOOKUP_FILTER
      AND t.LOOKUP_ID = s.LOOKUP_ID
);
```
